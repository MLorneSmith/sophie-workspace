## Implementation Steps

### 1. Create Theme Toggle Component

Create `apps/web/components/theme/theme-toggle.tsx`:

```typescript
'use client';

import { Moon, Sun } from 'lucide-react';
import { useTheme } from './use-theme';
import { cn } from '@kit/ui/cn';

export function ThemeToggle({ className }: { className?: string }) {
  const { theme, toggleTheme } = useTheme();

  return (
    <button
      onClick={toggleTheme}
      className={cn(
        'relative inline-flex h-9 w-9 items-center justify-center rounded-lg',
        'transition-colors hover:bg-gray-100 dark:hover:bg-gray-800',
        'focus:outline-none focus:ring-2 focus:ring-primary-500',
        className
      )}
      aria-label={`Switch to ${theme === 'light' ? 'dark' : 'light'} mode`}
      aria-pressed={theme === 'dark'}
    >
      <div className="relative h-5 w-5">
        <Sun
          className={cn(
            'absolute inset-0 h-5 w-5 transition-all',
            theme === 'light'
              ? 'rotate-0 scale-100'
              : 'rotate-90 scale-0'
          )}
        />
        <Moon
          className={cn(
            'absolute inset-0 h-5 w-5 transition-all',
            theme === 'dark'
              ? 'rotate-0 scale-100'
              : '-rotate-90 scale-0'
          )}
        />
      </div>
    </button>
  );
}
```

### 2. Create Standalone Toggle Version (for testing)

Create `apps/web/components/theme/theme-toggle-standalone.tsx`:

```typescript
'use client';

import { Moon, Sun } from 'lucide-react';
import { useState } from 'react';
import { cn } from '@kit/ui/cn';

export function ThemeToggleStandalone({ className }: { className?: string }) {
  const [isDark, setIsDark] = useState(false);

  return (
    <button
      onClick={() => setIsDark(!isDark)}
      className={cn(
        'relative inline-flex h-9 w-9 items-center justify-center rounded-lg',
        'transition-colors hover:bg-gray-100 dark:hover:bg-gray-800',
        'focus:outline-none focus:ring-2 focus:ring-primary-500',
        className
      )}
      aria-label={`Switch to ${isDark ? 'light' : 'dark'} mode`}
      aria-pressed={isDark}
    >
      <div className="relative h-5 w-5">
        <Sun
          className={cn(
            'absolute inset-0 h-5 w-5 transition-all duration-300',
            !isDark
              ? 'rotate-0 scale-100'
              : 'rotate-90 scale-0'
          )}
        />
        <Moon
          className={cn(
            'absolute inset-0 h-5 w-5 transition-all duration-300',
            isDark
              ? 'rotate-0 scale-100'
              : '-rotate-90 scale-0'
          )}
        />
      </div>
    </button>
  );
}
```

### 3. Add CSS Transitions

Create `apps/web/styles/theme-transitions.css`:

```css
/* Smooth theme transitions */
.theme-transition,
.theme-transition *,
.theme-transition *::before,
.theme-transition *::after {
  transition: background-color 300ms, border-color 300ms, color 300ms !important;
  transition-delay: 0ms !important;
}

/* Prevent transition on page load */
.no-transition,
.no-transition *,
.no-transition *::before,
.no-transition *::after {
  transition: none !important;
}
```
