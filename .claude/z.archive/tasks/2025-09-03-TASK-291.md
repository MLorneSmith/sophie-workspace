### Step 2: Smart Connection Logic

```bash
# Pseudo-code for smart-ssh.sh
resolve_and_connect() {
    local alias="$1"
    
    # 1. Resolve alias to actual codespace
    local codespace_info=$(resolve_codespace "$alias")
    
    # 2. Check codespace status
    local status=$(check_codespace_status "$codespace_info")
    
    # 3. Handle different states
    case "$status" in
        "Available")
            direct_ssh_connect "$codespace_info"
            ;;
        "Starting")
            wait_and_connect "$codespace_info"
            ;;
        "Stopped") 
            start_and_connect "$codespace_info"
            ;;
        "Missing"|"Renamed")
            update_alias_and_reconnect "$alias"
            ;;
        *)
            fallback_to_web "$codespace_info"
            ;;
    esac
}
```
